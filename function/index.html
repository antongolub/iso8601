    public parse(input: String, group: String, date: Date/string/number): Date/undefined source 

    import parse from '@antongolub/iso8601'

    Strict ISO 8601 date parser

    Params:

    NameTypeAttributeDescription
    input String
    group String
    • optional

    'date', 'time', 'datetime' are permissible; applies all if empty
    date Date/string/number
    • optional

    Ref date for 4.2.2 Local time representations. Defaults to new Date().

    Return:

    Date/undefined

    Example:

    parse('1970-01-01T00:00:00.000Z');         // new Date(0)
parse('1950-02');                          // new Date(1950, 1)
parse('1960W011');                         // new Date(1960, 0, 4)
parse('1950-205');                         // new Date(1950, 6, 24)
parse('1997-W01-2')                        // new Date(1996, 11, 31)
parse('+002015-W02-4');                    // new Date(2015, 0, 8);
parse('2052-05-09T15:20.5');               // new Date(2052, 4, 9, 15, 20, 30, 0);
